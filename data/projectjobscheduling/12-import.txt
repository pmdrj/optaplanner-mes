DECLARE
    c_time_scale CONSTANT NUMBER := 1440;
    p_scheduling_id NUMBER := 17;
    p_job_cnt       NUMBER;
    v_mode_nr       NUMBER;
    v_done_date     NUMBER;
    v_job_start     NUMBER;
    v_start_date    DATE;
BEGIN
    qmesp_bl_optaplanner.create_pjs_mm(p_scheduling_id => p_scheduling_id);
    SELECT COUNT(*) INTO p_job_cnt FROM qmes_opta_solution;

    FOR i IN 1 .. p_job_cnt
    LOOP
        dbms_output.put_line('job: ' || i);
    
        --execution mode    
        IF i IN (1, p_job_cnt)
        THEN
            v_mode_nr := 1;
        ELSE
            BEGIN
                SELECT j2r.mode_nr
                INTO   v_mode_nr
                FROM   qmes_opta_job2resource       j2r
                      ,qmes_opta_mach2res           m2r
                      ,qmes_opta_oper2job           o2j
                      ,qmes_ejs_scheduling_position sp
                WHERE  o2j.job_nr = j2r.job_nr
                AND    m2r.resource_nr = j2r.resource_nr
                AND    sp.machine_id = m2r.machine_id
                AND    sp.operation_id = o2j.operation_id
                AND    sp.scheduling_id = p_scheduling_id
                AND    j2r.job_nr = i;
            EXCEPTION
                WHEN no_data_found THEN
                    v_mode_nr := 1; --TODO: maszyna z obszaru roboczego
                WHEN too_many_rows THEN
                    qcmp_error.QRaise('QMES_EXC_AMBIGUOUS_MACHINE_MODE_ASSIGNMENT');
            END;
        END IF;
        dbms_output.put_line('mode: ' || v_mode_nr);
    
        --predecessors done date
        v_done_date := 0;
        FOR rec_pred IN (SELECT j2s.job_nr
                               ,Round(c_time_scale *
                                      (sp.stop_date - ss.begin_date)) AS done_date
                               ,ss.begin_date
                               ,sp.operation_id
                               ,sp.start_date
                               ,sp.stop_date
                         FROM   qmes_opta_job2successor      j2s
                               ,qmes_opta_oper2job           o2j
                               ,qmes_ejs_scheduling_position sp
                               ,qmes_ejs_scheduling_session  ss
                         WHERE  j2s.job_nr = o2j.job_nr
                         AND    sp.operation_id = o2j.operation_id
                         AND    sp.scheduling_id = ss.scheduling_id
                         AND    ss.scheduling_id = p_scheduling_id
                         AND    j2s.successor_nr = i)
        LOOP
            dbms_output.put_line('    predecessor: ' || rec_pred.job_nr);
            dbms_output.put_line('    operation: ' ||
                                 rec_pred.operation_id);
            dbms_output.put_line('    done date: ' || rec_pred.done_date || ' (' ||
                                 to_char(rec_pred.stop_date
                                        ,'yyyy-mm-dd hh24:mi') || ')');
        
            v_done_date := Greatest(v_done_date, rec_pred.done_date);
        END LOOP;
        dbms_output.put_line('predecessors done date: ' || v_done_date);
    
        --delay
        BEGIN
            SELECT sp.start_date
                  ,Round(c_time_scale * (sp.start_date - ss.begin_date)) AS job_start
            INTO   v_start_date
                  ,v_job_start
            FROM   qmes_ejs_scheduling_session  ss
                  ,qmes_ejs_scheduling_position sp
                  ,qmes_opta_oper2job           o2j
            WHERE  sp.scheduling_id = ss.scheduling_id
            AND    ss.scheduling_id = p_scheduling_id
            AND    sp.operation_id = o2j.operation_id
            AND    o2j.job_nr = i;
        EXCEPTION
            WHEN no_data_found THEN
                v_job_start  := -7;
                v_start_date := NULL;
            
            WHEN too_many_rows THEN
                qcmp_error.QRaise('QMES_EXC_AMBIGUOUS_OPERATION_JOB_ASSIGNMENT');
        END;
        dbms_output.put_line('start date: ' || v_job_start || ' (' ||
                             to_char(v_start_date, 'yyyy-mm-dd hh24:mi') || ')');
    
        dbms_output.put_line('');
    END LOOP;
END;
